declare_proto {
    fn print(msg: string) -> void {}
    fn putc(c: char) -> void {}
    fn printNumber(n: long) -> void {}
    fn readLine -> string {}
}

class Object {
	hash: long

	fn equals(other: Object) -> bool {
		=> false
	}

	fn to_string -> string {
		=> "Object"
	}

	fn to_hash -> long {
		=> hash
	}

	fn destruct {}
}


class Person: Object {
	age: int
	name: string

	hash: long

	fn to_string -> string {
		return "hello"
	}

	fn to_hash -> long {
		return 1
	}

	fn getAge -> int {
	 	return age
	}

	fn setAge (a: int) -> int {
		=> age = a
	}
}

fn main {
	var p: Person

	printNumber(p.setAge(3))
	putc('\n')
	printNumber(p.getAge())
	putc('\n')
	printNumber(p.age)
}
