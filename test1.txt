declare_proto {
    fn print(msg: string) -> void {}
    fn putc(c: char) -> void {}
    fn printNumber(n: long) -> void {}
    fn readLine -> string {}
}

class Object {
	hash: long
	fn construct () -> void {}
	fn destruct () -> void {}
	fn _op_equals (other: Object) -> bool => false
	fn to_string () -> string => "Object"
	fn to_hash () -> long => hash
}

class vec3 {
	x: long
	y: long
	z: long

	fn construct(x: long, y: long, z: long) {
		this.x = x
		this.y = y
		this.z = z
	}

	fn _op_add(o: vec3) -> vec3
		=> vec3(x + o.x, y + o.y, z + o.z)

}

fn main() {
	a := vec3(1, 2, 3)
	b := vec3(2, 3, 4)
	c := a + b
}
